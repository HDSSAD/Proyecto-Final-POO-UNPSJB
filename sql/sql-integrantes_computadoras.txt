-- DROP TABLE public.integrantes_computadoras;

CREATE TABLE public.integrantes_computadoras
(
    idcomputadora character varying COLLATE pg_catalog."default",
    idintegrante character varying COLLATE pg_catalog."default",
    serial integer NOT NULL DEFAULT nextval('integrantes_computadoras_serial_seq'::regclass),
    CONSTRAINT "integrantes_computadoras_idComputadora_key" UNIQUE (idcomputadora)
,
    CONSTRAINT fk_computadora FOREIGN KEY (idcomputadora)
        REFERENCES public.computadoras (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID,
    CONSTRAINT fk_integrante FOREIGN KEY (idintegrante)
        REFERENCES public.integrantes (dni) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
        NOT VALID
)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public.integrantes_computadoras
    OWNER to postgres;
