-- DROP TABLE public.integrantes;

CREATE TABLE public.integrantes
(
    dni character varying COLLATE pg_catalog."default" NOT NULL,
    apellido character varying COLLATE pg_catalog."default",
    nombre character varying COLLATE pg_catalog."default",
    fechanacimiento date,
    direccion character varying COLLATE pg_catalog."default",
    telefono character varying COLLATE pg_catalog."default",
    telefono2 character varying COLLATE pg_catalog."default",
    correo character varying COLLATE pg_catalog."default",
    "contraseña" character varying COLLATE pg_catalog."default",
    legajo smallint NOT NULL DEFAULT nextval('integrantes_legajo_seq'::regclass),
    tipo character varying COLLATE pg_catalog."default",
    CONSTRAINT integrantes_pkey PRIMARY KEY (legajo, dni),
    CONSTRAINT integrantes_dni_key UNIQUE (dni)

)
WITH (
    OIDS = FALSE
)
TABLESPACE pg_default;

ALTER TABLE public.integrantes
    OWNER to postgres;
