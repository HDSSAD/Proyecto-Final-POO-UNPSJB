    procesador character varying COLLATE pg_catalog."default",
    disco character varying COLLATE pg_catalog."default",
    ram character varying COLLATE pg_catalog."default",
    lectora character varying COLLATE pg_catalog."default",
    notas character varying COLLATE pg_catalog."default",
    CONSTRAINT computadoras_pkey PRIMARY KEY (id, serialcomputadora),
    CONSTRAINT "computadoras_idComputadora_key" UNIQUE (id)

    OIDS = FALSE
